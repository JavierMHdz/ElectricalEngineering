D = 1000;

%Graph 1 - Speed = 0.1*c
figure

%Traveling Twin
for x = 0:D
    ct = 0.1*x;
    plot(x,ct);
    hold on

    ct = (-0.1*x)+(2*0.1*D);
    plot(x,ct)
    hold on
    grid on
end


%Earth-bound Twin
for ct = 0:5:(2*0.1*D)
    x = 0;
    plot(x,ct,'r*');
    hold on;
end

%Outbound rocket ship
for x = 0:100:2*D
    ct = 0.1*x;
    plot(x,ct,'o');
    hold on
end

%Inbound rocket ship
for x = 0:100:2*D
    ct = (-0.1*x)+(2*0.1*D);
    plot(x,ct,'k+')
    hold on
end
title('Twin Paradox at 0.1*c');
xlabel('x');
ylabel('ct');
